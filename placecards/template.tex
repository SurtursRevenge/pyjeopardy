%
% Parameters:
% 1: number
% 2: background color
% 3: font color
% 4: color name
%
\newcommand*\cardfront[4]{%
    \definecolor{bgcolor}{HTML}{#2}%
    \definecolor{fontcolor}{HTML}{#3}%
    \color{fontcolor}%
    \begin{tikzpicture}[x=1cm, y=1cm, inner sep=0, outer sep=0, anchor=north west]
        % background
        \node (bg) at (0,0) [fill=bgcolor,minimum width=\cardwidth,minimum height=\cardheight] {};

        % text
        \begin{scope}[shift={(bg.north west)}, x={(bg.north)}, y={(bg.west)}, scale=2]
            \node at (0.05,0.35) {\Huge\##1};

            \node[rotate=90, text width=\cardheight, align=center] at (0.9,1) {\small #4};
        \end{scope}
    \end{tikzpicture}%
}
